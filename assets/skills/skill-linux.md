### 2.2 chmod - 修改权限

chmod o=r file.txt        # 其他用户只读

chmod a+r file.txt        # 所有人可读

chgrp groupname file.txt         # 仅修改组

### 2.4 sudo - 提升权限

### 8.3 条件判断

if [ $a -gt $b ]; then
    echo "a > b"
elif [ $a -lt $b ]; then
    echo "a < b"
else
    echo "a == b"
fi

# 常用比较:
# -eq 等于  -ne 不等于  -gt 大于  -lt 小于  -ge 大于等于  -le 小于等于
# -z 字符串为空  -n 字符串非空
# -f 文件存在  -d 目录存在  -r 可读  -w 可写  -x 可执行

### 8.4 循环

# for 循环

    echo $i
done

for file in *.txt; do
    echo "Found: $file"

# while 循环
count=0
while [ $count -lt 5 ]; do
    echo $count
    count=$((count + 1))

# case 语句
case $var in
    start) echo "Starting..." ;;
    stop)  echo "Stopping..." ;;
    *)     echo "Unknown" ;;
esac
